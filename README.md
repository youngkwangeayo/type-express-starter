npx tsc --init              # tsconfig.json 생성

npm install ts-node-dev     # ts파일 실시간 실행 (도중컴파일불필요) nodemon 파일감지기능  ts-node + nodemon 합쳐놓은 개발 툴
npm install @types/node     # js 라이브러리 타입파일 알기 편하게
